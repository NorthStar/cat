  background.position = @page.evaluate_script(%Q{$("#{@selector}").css("backgroundPosition")})

def margin
  margins = Dimension.new
  margins.top    = @page.evaluate_script(%Q{$("#{@selector}").css("marginTop")}).to_i
  margins.right  = @page.evaluate_script(%Q{$("#{@selector}").css("marginRight")}).to_i
  margins.bottom = @page.evaluate_script(%Q{$("#{@selector}").css("marginBottom")}).to_i
  margins.left   = @page.evaluate_script(%Q{$("#{@selector}").css("marginLeft")}).to_i

  margins
end
